rules_version = '2';
service firebase.storage {
  match /b/{bucket}/o {

    match /profile_photos/{userId}/{allPaths=**} {
      allow write: if request.auth != null
                   && request.auth.uid == userId
                   && request.resource.contentType.matches('image/(jpeg|png)');
      allow read: if request.auth != null;
    }

    // Uzman: sertifikalar (görsel veya PDF)
    match /certificates/{userId}/{allPaths=**} {
      allow write: if request.auth != null
                   && request.auth.uid == userId
                   && (
                     request.resource.contentType.matches('image/(jpeg|png|heic)')
                     || request.resource.contentType == 'application/pdf'
                   );
      allow read: if request.auth != null;
    }

    // Uzman: kimlik / doğrulama belgeleri (ön–arka yüz, selfie vb.)
    match /verification_documents/{userId}/{allPaths=**} {
      allow write: if request.auth != null
                   && request.auth.uid == userId
                   && request.resource.contentType.matches('image/(jpeg|png)');
      allow read: if request.auth != null;
    }

    // Servis görselleri
    match /service_images/{allPaths=**} {
      allow read: if request.auth != null;
      allow write: if false;
    }

    // Uzman: portföy (önceki işlerden fotoğraflar)
    match /portfolio/{userId}/{allPaths=**} {
      allow write: if request.auth != null
                   && request.auth.uid == userId
                   && request.resource.contentType.matches('image/(jpeg|png|heic)');
      allow read: if request.auth != null;
    }

    // Uzman: ilan görseli (ilan açma/düzenleme)
    match /listing_images/{userId}/{allPaths=**} {
      allow write: if request.auth != null
                   && request.auth.uid == userId
                   && request.resource.contentType.matches('image/(jpeg|png|heic)');
      allow read: if request.auth != null;
    }
  }
}
